251  template<typename... I, typename = typename std::enable_if<sizeof...(I) == NumDims>::type>
252  ManagedGrid<Dtype, NumDims> resized(I... sizes) {
253  cpu_grid_t g(nullptr, sizes...);
254  if(g.size() <= capacity) {
255  //no need to allocate and copy; capacity stays the same
256  ManagedGrid<Dtype, NumDims> tmp;
257  tmp.cpu_ptr = cpu_ptr;
258  tmp.gpu_info = gpu_info;
259  tmp.cpu_grid = cpu_grid_t(cpu_ptr.get(), sizes...);
260  if(gpu_info) tmp.gpu_grid = gpu_grid_t(gpu_info->gpu_ptr, sizes...);
261  tmp.capacity = capacity;
262  return tmp;
263  } else {
264  ManagedGrid<Dtype, NumDims> tmp(sizes...);
265  if(size() > 0 && tmp.size() > 0) {
266  if(ongpu()) tmp.togpu(); //allocate gpu memory
267  copyTo(tmp);
268  }
269  return tmp;
270  }

285  void togpu(bool dotransfer=true) const {
286  if(capacity == 0) return;
287  //check that memory is allocated - even if data is on gpu, may still need to set this mgrid's gpu_grid
288  if(gpu_grid.data() == nullptr) {
289  if(gpu_info->gpu_ptr == nullptr) {

293  gpu_grid.set_buffer(gpu_info->gpu_ptr+offset);
294  }
