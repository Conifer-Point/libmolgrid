155  ReceptorStratifiedExampleRefProvider(): currenti(0), currentk(0), randomize(false) {}

157  ExampleRefProvider(settings), param(settings), currenti(0), currentk(0), randomize(settings.shuffle)

161  void addref(const ExampleRef& ex)
162  {
163  if(ex.files.size() == 0) {
164  throw std::invalid_argument("Missing receptor from line");
165  }
166  if(recmap.count(ex.files[0]) == 0)
167  {
168  //allocate
169  recmap[ex.files[0]] = examples.size();
170  examples.push_back(Provider(param));
171  }
172  unsigned pos = recmap[ex.files[0]];
173  examples[pos].addref(ex);
174  }

177  for(unsigned i = 0, n = examples.size(); i < n; i++) {
178  if(examples[i].size() > 0)
179  return examples[i].num_labels();
180  }
181  return 0;
182  }
183 
184  //note there is an explicit specialization for balanced providers, k=2
185  void setup()
186  {
187  if(K <= 0) throw std::invalid_argument("Invalid sampling k for ReceptorStratifiedExampleRefProvider");
188  currenti = 0; currentk = 0;
189 
190  for(unsigned i = 0, n = examples.size(); i < n; i++)
191  {
192  examples[i].setup();
193  }
194  //also shuffle receptors
195  if(randomize) shuffle(examples.begin(), examples.end(), random_engine);
196  }
197 
198  void nextref(ExampleRef& ex)
199  {
200  if(examples.size() == 0) throw std::invalid_argument("No valid stratified examples.");
201  if(currentk >= K)
202  {
203  currentk = 0; //on to next receptor
204  currenti++;
205  }
206  if(currenti >= examples.size())
207  {
208  currenti = 0;
209  if(currentk != 0) std::logic_error("Invalid indices");
210  if(randomize) shuffle(examples.begin(), examples.end(), random_engine);
211  }
212 
213  if(examples[currenti].size() == 0) throw std::logic_error("No valid sub-stratified examples.");
214  examples[currenti].nextref(ex);
215  currentk++;
216  }
217 
218  unsigned size() const
219  {
220  //no one said this had to be particularly efficient..
221  unsigned ret = 0;
222  for(unsigned i = 0, n = examples.size(); i < n; i++)
223  {
224  ret += examples[i].size();
225  }
226  return ret;
227  }

238 class ValueStratifiedExampleRefProfider: public ExampleRefProvider
239 {
240  std::vector<Provider> examples;
241  size_t currenti = 0;//position in array

245 
246  //return bin for given value
247  unsigned bin(double val) const
248  {
249  if(use_abs)
250  val = fabs(val);
251  if(val < min) val = min;
252  if(val >= max) val = max-FLT_EPSILON;
253  val -= min;
254  unsigned pos = val/step;
255  return pos;
256  }

259  ValueStratifiedExampleRefProfider(const ExampleProviderSettings& parm):
260  ExampleRefProvider(parm), currenti(0)

262  max = parm.stratify_max;
263  min = parm.stratify_min;
264  step = parm.stratify_step;
265  use_abs = parm.stratify_abs;
266  valpos = parm.stratify_pos;
267 
268  if(valpos < 0) throw std::invalid_argument("Negative position for stratification value");
269  if(min == max) throw std::invalid_argument("Empty range for value stratification");
270  unsigned maxbin = bin(max);

275  examples.push_back(Provider(parm));

393  void nextref(ExampleRef& ex) {

395  current_group_index = 0; //wrap and start next time step

399  //wrap around

423  }

428  return examples.size();
429  }
