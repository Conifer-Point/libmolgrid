309 class MappedAtomIndexTyper: public AtomIndexTyper {
310  protected:
311  Mapper mapper;
312  Typer typer;
313 
314  std::vector<float> type_radii;
315  public:
316  MappedAtomIndexTyper(const Mapper& map, const Typer& typr): mapper(map), typer(typr) {
317  unsigned oldN = typer.num_types();
318  unsigned newN = mapper.num_types();
319 
320  std::vector< std::vector<float> > radii(newN);
321  for(unsigned ot = 0; ot < oldN; ot++) {
322  auto t_r = typer.get_int_type(ot);
323  if(t_r.first >= 0) {
324  unsigned nt = mapper.get_new_type(t_r.first);
325  if(nt < radii.size()) {
326  radii[nt].push_back(t_r.second);
327  }
328  }
329  }
330 
331  type_radii.resize(newN);
332  for(unsigned i = 0; i < newN; i++) {
333  float sum = 0.0;
334  for(unsigned j = 0, n = radii[i].size(); j < n; j++) {
335  sum += radii[i][j];
336  }
337  type_radii[i] = sum/radii[i].size();
338  }
339  }
340 
341  virtual ~MappedAtomIndexTyper() {}
342 
344  virtual unsigned num_types() const {
345  return mapper.num_types();
346  }
347 
349  virtual std::pair<int,float> get_atom_type_index(OpenBabel::OBAtom* a) const {
350  auto res_rad = typer.get_atom_type_index(a);
351  //remap the type
352  int ret = mapper.get_new_type(res_rad.first);
353  return std::make_pair(ret, res_rad.second);
354  }
355 
356  //map the type
357  virtual std::pair<int,float> get_int_type(int t) const {
358  auto res_rad = typer.get_int_type(t);
359  //remap the type
360  int ret = mapper.get_new_type(res_rad.first);
361  return std::make_pair(ret, res_rad.second);
362  }
363 
364  //return vector of string representations of types
365  virtual std::vector<std::string> get_type_names() const {
366  return mapper.get_type_names();
367  }
368 
370  virtual std::vector<float> get_type_radii() const { return type_radii; }
