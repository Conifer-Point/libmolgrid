169  /* \brief Generate grid tensor from a vector of examples, as provided by ExampleProvider.next_batch.
170  * Coordinates may be optionally translated/rotated. Do not use this function
171  * if it is desirable to retain the transformation used (e.g., when backpropagating).
172  * The center of the last coordinate set before transformation
173  * will be used as the grid center.
174  *
175  * @param[in] ex example
176  * @param[in] transform transformation to apply
177  * @param[out] out a 4D grid
178  * @param[in] random_translation maximum amount to randomly translate each coordinate (+/-)
179  * @param[in] random_rotation whether or not to randomly rotate
180  */
181  template <typename Dtype, bool isCUDA>
182  void forward(const std::vector<Example>& in, Grid<Dtype, 5, isCUDA>& out, float random_translation=0.0, bool random_rotation = false) const {
183  if(in.size() != out.dimension(0)) throw std::out_of_range("output grid dimension does not match size of example vector");
184  for(unsigned i = 0, n = in.size(); i < n; i++) {
185  Grid<Dtype, 4, isCUDA> g(out[i]);
186  forward<Dtype,isCUDA>(in[i],g, random_translation, random_rotation);
187  }
188  }
